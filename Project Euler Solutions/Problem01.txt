public class Problem01 {
	public static void main(String args[]) {
		int sum = 0;
		for (int i = 1; i < 1000; i++) {
			if (isMultiple(i))
				sum += i;
		}
		System.out.println("Answer: " + sum);
	}
	private static boolean isMultiple(int num) {
		if (num % 5 == 0 || num % 3 == 0) {
			return true;
		}
		return false;
	}
}

Answer: 233168
